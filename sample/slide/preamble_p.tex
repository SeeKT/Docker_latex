  \edef\pgf@temp{\the\wd\pgfutil@abb}%

    \unhbox\pgfutil@abb%
    \hskip\dimexpr2in-2\hoffset-\pgf@temp\relax% changed

    \hskip\dimexpr-2in-2\hoffset\relax% new
  }%
